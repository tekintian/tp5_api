206:      * 调用反射执行类的方法 支持参数绑定

212:     public static function invokeMethod($method, $vars = [])

246:     /**
247:      * 绑定参数
248:      * @access public
249:      * @param \ReflectionMethod|\ReflectionFunction $reflect 反射类
250:      * @param array                                 $vars    变量
251:      * @return array
252:      */
253:     private static function bindParams($reflect, $vars = [])
254:     {
255:         if (empty($vars)) {
256:             // 自动获取请求变量
257:             if (Config::get('url_param_type')) {
258:                 $vars = Request::instance()->route();
259:             } else {
260:                 $vars = Request::instance()->param();
261:             }
262:         }
263:         $args = [];
264:         // 判断数组类型 数字数组时按顺序绑定参数
265:         reset($vars);
266:         $type = key($vars) === 0 ? 1 : 0;
267:         if ($reflect->getNumberOfParameters() > 0) {
268:             $params = $reflect->getParameters();
269:             foreach ($params as $param) {
270:                 $name  = $param->getName();
271:                 $class = $param->getClass();
272:                 if ($class) {
273:                     $className = $class->getName();
274:                     $bind      = Request::instance()->$name;
275:                     if ($bind instanceof $className) {
276:                         $args[] = $bind;
277:                     } else {
278:                         if (method_exists($className, 'invoke')) {
279:                             $method = new \ReflectionMethod($className, 'invoke');
280:                             if ($method->isPublic() && $method->isStatic()) {
281:                                 $args[] = $className::invoke(Request::instance());
282:                                 continue;
283:                             }
284:                         }
285:                         $args[] = method_exists($className, 'instance') ? $className::instance() : new $className;
286:                     }
287:                 } elseif (1 == $type && !empty($vars)) {
288:                     $args[] = array_shift($vars);
289:                 } elseif (0 == $type && isset($vars[$name])) {
290:                     $args[] = $vars[$name];
291:                 } elseif ($param->isDefaultValueAvailable()) {
292:                     $args[] = $param->getDefaultValue();
293:                 } else {
294:                     throw new \InvalidArgumentException('method param miss:' . $name);
295:                 }
296:             }
297:         }
298:         return $args;
299:     }

301:     /**
302:      * 执行模块
303:      * @access public
304:      * @param array $result 模块/控制器/操作
305:      * @param array $config 配置参数
306:      * @param bool  $convert 是否自动转换控制器和操作名
307:      * @return mixed

309:     public static function module($result, $config, $convert = null)
310:     {
311:         if (is_string($result)) {
312:             $result = explode('/', $result);
313:         }
314:         $request = Request::instance();
315:         if ($config['app_multi_module']) {
316:             // 多模块部署
317:             $module    = strip_tags(strtolower($result[0] ?: $config['default_module']));
318:             $bind      = Route::getBind('module');
319:             $available = false;
320:             if ($bind) {
321:                 // 绑定模块
322:                 list($bindModule) = explode('/', $bind);
323:                 if (empty($result[0])) {
324:                     $module    = $bindModule;
325:                     $available = true;
326:                 } elseif ($module == $bindModule) {
327:                     $available = true;
328:                 }
329:             } elseif (!in_array($module, $config['deny_module_list']) && is_dir(APP_PATH . $module)) {
330:                 $available = true;
331:             }
332: 
333:             // 模块初始化
334:             if ($module && $available) {
335:                 // 初始化模块
336:                 $request->module($module);
337:                 $config = self::init($module);
338:                 // 模块请求缓存检查
339:                 $request->cache($config['request_cache'], $config['request_cache_expire']);
340:             } else {
341:                 throw new HttpException(404, 'module not exists:' . $module);
342:             }
343:         } else {
344:             // 单一模块部署
345:             $module = '';
346:             $request->module($module);
347:         }
348:         // 当前模块路径
349:         App::$modulePath = APP_PATH . ($module ? $module . DS : '');
350: 
351:         // 是否自动转换控制器和操作名
352:         $convert = is_bool($convert) ? $convert : $config['url_convert'];
353:         // 获取控制器名
354:         $controller = strip_tags($result[1] ?: $config['default_controller']);
355:         $controller = $convert ? strtolower($controller) : $controller;
356: 
357:         // 获取操作名
358:         $actionName = strip_tags($result[2] ?: $config['default_action']);
359:         $actionName = $convert ? strtolower($actionName) : $actionName;
360: 
361:         // 设置当前请求的控制器、操作
362:         $request->controller(Loader::parseName($controller, 1))->action($actionName);
363: 
364:         // 监听module_init
365:         Hook::listen('module_init', $request);
366: 
367:         $instance = Loader::controller($controller, $config['url_controller_layer'], $config['controller_suffix'], $config['empty_controller']);
368:         if (is_null($instance)) {
369:             throw new HttpException(404, 'controller not exists:' . Loader::parseName($controller, 1));
370:         }
371:         // 获取当前操作名
372:         $action = $actionName . $config['action_suffix'];
373: 
374:         $vars = [];
375:         if (is_callable([$instance, $action])) {
376:             // 执行操作方法
377:             $call = [$instance, $action];
378:         } elseif (is_callable([$instance, '_empty'])) {
379:             // 空操作
380:             $call = [$instance, '_empty'];
381:             $vars = [$actionName];
382:         } else {
383:             // 操作不存在
384:             throw new HttpException(404, 'method not exists:' . get_class($instance) . '->' . $action . '()');
385:         }
386: 
387:         Hook::listen('action_begin', $call);
388: 
389:         return self::invokeMethod($call, $vars);
390:     }
